htpasswd is used to create and update the flat-files used to store usernames and password for basic authentication of HTTP users. 

htpasswd -c -B -b passwdfile username password

ex:
htpasswd -c -B -b htpasswdfile user1 password1
htpasswd -B -b htpasswdfile user2 password2

#Create a secret in the openshift-config namespace using the HTPasswd file
oc create secret generic  htpass-secret --from-file htpasswd=/path/to/htpasswdfile -n openshift-config

#Configure the OpenShift OAuth server to use the HTPasswd identity provider:
oc edit oauth cluster

#Add the following configuration under spec.identityProviders:
    - name: my_htpasswd_provider
      challenge: true
      login: true
      mappingMethod: claim
      type: HTPasswd
      htpasswd:
        fileData:
          name: htpass-secret


#or in console

#update new password - old password will be overwritten by new password
htpasswd -b htpasswdfile user1 oldpassword
htpasswd -b htpasswdfile user1 newpassword

#Delete credential
htpasswd -D htpasswdfile user2

also delete user from oauth after updating the secret, even if secret is updated to remove user creds user will remain in cluster
oc delete user user2 

#Update a secret in the openshift-config namespace after changes in the HTPasswd file
oc set data secret htpass-secret --from-fie htpasswd=/path/to/htpasswdfile -n openshift-config

#assign cluster admin role privileges to user
oc adm policy add-cluster-role-to-user cluster-admin user1
